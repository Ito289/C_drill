\subsection*{問1}

{\ttfamily while}文を用いたプログラムに書き直せ。
\begin{verbatim}
int r = -4;
printf("%d ", r);
r = r + 2;
printf("%d ", r);
r = r + 2;
printf("%d ", r);
r = r + 2;
printf("\n");
\end{verbatim}

\ifnum \anss=1
\begin{verbatim}






\end{verbatim}
\else
\begin{verbatim}
int r = -4;
while(r < 1){  または (r<2) または (r<=0)
  printf("%d ", r);
  r = r + 2;
}
printf("\n");
\end{verbatim}
\fi
