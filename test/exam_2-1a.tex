\subsection*{問2}

次のプログラム終了時には何が表示されるか。
\begin{verbatim}
char b1, b2;
b1 = 'k' - 'K' + 'J';
b2 = 'm' - 4;
printf("%d %c\n", b1, b1);
printf("%d %c\n", b2, b2);
\end{verbatim}

\ifnum \anss=1
\vspace*{1.5cm}
\else
\begin{verbatim}
106 j
105 i
\end{verbatim}
\fi


\subsection*{問3}

次のプログラム終了時には何が表示されるか。
\begin{verbatim}
int a[3] = {2, 3, 1}, x = 1, b[3] = {5, 4, 8};
printf("%d %d %d\n", a[a[0]], a[1], a[2]);
printf("%d %d %d\n", b[a[0]-2], b[a[1]-3], b[a[2]-1]);
printf("%d %d %d\n", b[x + 1], b[x], b[x - 1]);
\end{verbatim}

\ifnum \anss=1
\vspace*{1.5cm}
\else
\begin{verbatim}
1 3 1
5 5 5
8 4 5
\end{verbatim}
\fi
